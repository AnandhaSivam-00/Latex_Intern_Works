%               For Miniprojects

\fontsize{\textwidth}{\textwidth}
\newcommand{\vizappear}[2]{
    \adjustbox{valign=t}{\begin{minipage}{0.4\textwidth}
    \noindent \hangafter=0
       \hspace*{0.2cm} \textmd{#1}
    \end{minipage}}
    \adjustbox{valign=t}{\begin{minipage}{0.75\textwidth}
     \vspace*{0.05cm}
     \noindent \hangafter=0
     \colorbars{#2}
    \end{minipage}} \smallskip
}

\newcommand{\vizskills}[4]{
    \setstretch{0}
    \noindent \textbf{#1} \\ \vspace*{0.35cm}

    \def \groupOfSkills{#2}

    \def \skillnames{#3}

    \def \skillrate{#4}

    \foreach \i in {0,...,\inteval{\groupOfSkills-1}}{
        \pgfmathsetmacro{\skillA}{{\skillnames}[\i][0]}
        \pgfmathsetmacro{\skillB}{{\skillnames}[\i][1]}
        \pgfmathsetmacro{\skillC}{{\skillnames}[\i][2]}
        
        \pgfmathsetmacro{\skillApercent}{{\skillrate}[\i][0]}
        \pgfmathsetmacro{\skillBpercent}{{\skillrate}[\i][1]}
        \pgfmathsetmacro{\skillCpercent}{{\skillrate}[\i][2]}

        \begin{minipage}{0.3\textwidth}
           \vizappear{\skillA}{\skillApercent}
        \end{minipage}
        \begin{minipage}{0.03\textwidth}
            \hfill
        \end{minipage}
        \begin{minipage}{0.3\textwidth}
            \ifnum \skillBpercent > 0
                \vizappear{\skillB}{\skillBpercent}
            \fi
        \end{minipage}
        \begin{minipage}{0.03\textwidth}
            \hfill
        \end{minipage}
        \begin{minipage}{0.3\textwidth}
           \ifnum \skillCpercent > 0
                \vizappear{\skillC}{\skillCpercent}
            \fi
        \end{minipage}
    }
}
